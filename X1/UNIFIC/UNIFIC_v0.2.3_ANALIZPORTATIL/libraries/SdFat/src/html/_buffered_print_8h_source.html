234 bool sync() {
235 if (!m_wr || m_wr->write(m_buf, m_in) != m_in) {
236 return false;
237 }
238 m_in = 0;
239 return true;
240 }
249 size_t write(const void* src, size_t n) {
250 if ((m_in + n) > sizeof(m_buf)) {
251 if (!sync()) {
252 return 0;
253 }
254 if (n >= sizeof(m_buf)) {
255 return n == m_wr->write((const uint8_t*)src, n) ? n : 0;
256 }
257 }
258 memcpy(m_buf + m_in, src, n);
259 m_in += n;
260 return n;
261 }
